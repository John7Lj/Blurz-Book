.model small
.stack 100h
.data 

message db 'the printed numbres from 1 to 100 here $' 
line db 0ah,0dh, '$' 

adding1 db 30h
adding2 db 30h

.code 

main proc far 

.startup 

mov ah ,09
lea dx , message
int 21h

mov cl, -1          ; start at -1 (so first INC makes it 0)

outer_print:
    inc cl          ; CL = 0..9
    cmp cl, 10
    jge exit        ; stop AFTER CL reaches 10

    mov bl, 0

inner_print:
    mov ah, 02h

    mov dl, adding1
    add dl, cl      ; tens digit
    int 21h

    mov dl, adding2
    add dl, bl      ; ones digit
    int 21h

    mov ah, 09h
    lea dx, line
    int 21h

    inc bl
    cmp bl, 10
    jne inner_print

    jmp outer_print

exit:
    mov ax, 4C00h
    int 21h

main endp  
end main

The issue is that you're using **both** `.startup` and `main PROC far` together, which causes a conflict. Let me show you two correct ways to fix this:

**Option 1: Remove `.startup` and add proper initialization**

```assembly
.model small
.stack 100h

.data 
 table db 22, 33, 'k'

.code
main PROC far 
    mov ax, @data       ; Initialize data segment
    mov ds, ax          ; DS must point to data segment
    
    mov al, table[2]  
    mov dl, al 
    mov ah, 02
    int 21h
    
    mov ax, 4C00h
    int 21h                            
main endp 
end main
```

**Option 2: Use `.startup` properly (remove `main PROC far`)**

```assembly
.model small
.stack 100h

.data 
 table db 22, 33, 'k'

.code
.startup

    mov al, table[2]  
    mov dl, al 
    mov ah, 02
    int 21h

.exit
end
```

.code

    mov ax,@data
    mov ds,ax

    lea dx, message
    mov ah, 09h
    int 21h            ; print "plz enter digit"

    mov ah, 01h
    int 21h            ; AL = ASCII char typed, e.g. '5' (35h)

    lea dx, message2
    mov ah, 09h
    int 21h            ; print "you enter "

    mov dl, al         ; DL = ASCII digit (already correct)
    mov ah, 02h
    int 21h            ; print the single character

    mov ax, 4C00h
    int 21h            ; exit
main ENDP
end main
